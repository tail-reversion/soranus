% Date Parsing
%

\regex_const:Nn \g__soranus_iso_datetime_regex
  {
    \A % must be start of tl
      \d{4} % year
      (:?
        -[0-1]\d % month
        (:?
          -[0-2]\d % day
          (:?
            T[0-2]\d % hour
            (:?
              :[0-5]\d % minute
              (:?
                :[0-5]\d % second
                Z? % UTC
              )?
            )?
          )?
        )?
      )?
    \Z % must be end of tl
  }

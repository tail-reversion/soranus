\keys_define:nn { soranus / class-options }
  {
    paper .meta:nn = { soranus / document-setup } { paper = #1 },

    point-size .meta:nn = { soranus / document-setup } { point-size = #1 },

    landscape .meta:nn           = { soranus / document-setup } { landscape },
    landscape .value_forbidden:n = true,

    two-side .meta:nn           = { soranus / document-setup } { two-side },
    two-side .value_forbidden:n = true
  }


\NewDocumentCommand \DocumentSetup { m }
  {
    \keys_set:nn { soranus / document-setup } { #1 }
    \__soranus_init_document_point_size:
    \__soranus_init_paper_and_trim:
  }
